% -------------------- CUSTOM BLOCKS --------------------

% Inline hyperlink shown in linkBlue (used for advisor names, project pages, etc.).
\newcommand{\cvLink}[2]{\href{#1}{\color{linkBlue}#2}}

% Two-row heading: institution / dates  +  role / detail.
% Uses \hfill so the title takes whatever width it needs (up to the full line)
% with the right-side annotation flush right.  \needspace keeps the heading
% from being orphaned at the foot of a page.
\newcommand{\cvHeading}[4]{%
  \needspace{4\baselineskip}%
  \vspace{1pt}%
  \noindent
  {\color{accent}\bfseries #1}\hfill{\color{muted}\small #2}\par
  \vspace{1pt}%
  \noindent
  {\itshape\color{ink} #3}\hfill{\color{muted}\small\itshape #4}\par
  \vspace{1pt}%
}

% Project block: bold title with optional right-aligned date.  When no date
% is provided, the title takes the full line width (so long titles don't wrap
% prematurely).
\newcommand{\cvProject}[2]{%
  \vspace{2pt}%
  \notblank{#2}{%
    \noindent{#1}\hfill{\color{muted}\small #2}\par
  }{%
    \noindent{#1}\par
  }%
  \vspace{0pt}%
}

% Project block with a links sub-line (rendered in small font, slightly
% indented).  Useful when project pages / reports / code links accompany
% a long title.
\newcommand{\cvProjectLinks}[2]{%
  \vspace{2pt}%
  {#1}%
  \notblank{#2}{\\[1pt]\hspace*{1.2em}{\small #2}}{}%
  \par\vspace{0pt}%
}

% Small icon badge: \cvBadge{url}{bg-color}{icon-cmd}
\newcommand{\cvBadge}[3]{%
  \href{#1}{\tikz[baseline=(b.base)]\node[
    inner xsep=3pt, inner ysep=0.6pt,
    rounded corners=1.5pt, fill=#2, text=white,
    font=\sffamily\scriptsize] (b) {#3};}%
}
% Variant with custom text colour (for light backgrounds, e.g. HF yellow)
\newcommand{\cvBadgeT}[4]{%
  \href{#1}{\tikz[baseline=(b.base)]\node[
    inner xsep=3pt, inner ysep=0.6pt,
    rounded corners=1.5pt, fill=#2, text=#3,
    font=\sffamily\bfseries\scriptsize] (b) {#4};}%
}

\newcommand{\cvLinkPDF}[1]{\cvBadge{#1}{badge}{\faIcon{file-pdf}}}
\newcommand{\cvLinkCode}[1]{\cvBadge{#1}{badge}{\faIcon{github}}}
\newcommand{\cvLinkHome}[1]{\cvBadge{#1}{badge}{\faIcon{globe}}}
% Hugging Face: real 🤗 emoji image (pdflatex can't render emoji glyphs natively).
\newcommand{\cvLinkHF}[1]{%
  \href{#1}{\raisebox{-0.15em}{\includegraphics[height=1.05em]{assets/hf_face.png}}}%
}

% GitHub badge with star count: a two-segment connected pill.
% \cvCodeStar{url}{star-count}
\newcommand{\cvCodeStar}[2]{%
  \href{#1}{\tikz[baseline=(L.base)]{%
    \node[fill=ghDark, text=white,
          font=\sffamily\scriptsize,
          inner xsep=3pt, inner ysep=0.6pt,
          rounded corners=1.5pt,
          outer sep=0pt] (L) {\faIcon{github}};
    \node[fill=ghLight, text=ghStar,
          font=\sffamily\bfseries\scriptsize,
          inner xsep=3pt, inner ysep=0.6pt,
          rounded corners=1.5pt,
          outer sep=0pt,
          anchor=west, xshift=-2pt] at (L.east) (R) {\faIcon{star}\,#2};
  }}%
}

% Publication entry (9 args):
%   #1 title  #2 authors  #3 venue  #4 status
%   #5 pdf-url  #6 code-url  #7 homepage-url  #8 huggingface-url
%   #9 github star count (renders two-segment badge if non-empty AND #6 set)
% Pass empty {} for anything that doesn't apply.
\newcommand{\cvPub}[9]{%
  \item[\textcolor{accent}{\small\faAngleDoubleRight}]%
    \textbf{#1}%
    \notblank{#5}{~\cvLinkPDF{#5}}{}%
    \notblank{#6}{\notblank{#9}{~\cvCodeStar{#6}{#9}}{~\cvLinkCode{#6}}}{}%
    \notblank{#7}{~\cvLinkHome{#7}}{}%
    \notblank{#8}{~\cvLinkHF{#8}}{}%
    \\[1pt]
    {\footnotesize\color{ink} #2}\\[1pt]
    {\footnotesize\itshape\color{accentSoft} #3}%
    \notblank{#4}{\,\,{\footnotesize\color{accent}\textbf{[#4]}}}{}%
}

% Inline tag (for skills, etc.)
\newcommand{\cvTag}[1]{%
  \tikz[baseline=(t.base)]\node[
    inner xsep=5pt, inner ysep=1.5pt,
    rounded corners=2pt,
    fill=tagBg, text=accent,
    font=\sffamily\footnotesize] (t) {#1};\,%
}

% Skill row: \cvSkillRow{Languages}{C/C++, Python, …}
\newcommand{\cvSkillRow}[2]{%
  \vspace{1pt}%
  \noindent
  \begin{minipage}[t]{0.16\linewidth}
    \raggedright{\rmfamily\bfseries\color{accent}\footnotesize\MakeUppercase{#1}}
  \end{minipage}\hfill
  \begin{minipage}[t]{0.83\linewidth}
    \raggedright #2
  \end{minipage}%
  \par%
}
